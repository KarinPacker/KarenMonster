    <script>
        // --- CONFIGURATION ---
        let currentXP = 0;
        let streak = 0;
        let lastLogDate = "";

        // --- LOAD DATA ---
        if(localStorage.getItem('codemonXP')) {
            currentXP = parseInt(localStorage.getItem('codemonXP'));
            streak = parseInt(localStorage.getItem('codemonStreak')) || 0;
            lastLogDate = localStorage.getItem('codemonLastDate') || "";
        }

        // --- UPDATE UI ---
        const updatePetVisuals = () => {
            const petDisplay = document.getElementById('pet-display');
            const nameDisplay = document.getElementById('pet-name');
            const progressFill = document.getElementById('progress-fill');
            const xpDisplay = document.getElementById('xp-display');
            const streakDisplay = document.getElementById('streak-display');

            xpDisplay.innerText = Math.floor(currentXP);
            streakDisplay.innerText = streak;

            // Progress Bar Logic (Scales every 1000 XP)
            let percentage = (currentXP % 1000) / 10; 
            // If we are over 1000, bar fills based on modulo
            if(percentage === 0 && currentXP > 0) percentage = 100;
            progressFill.style.width = percentage + '%';

            // Evolution Logic
            if (currentXP < 100) {
                petDisplay.innerText = "ðŸ¥š";
                nameDisplay.innerText = "Level 1: The Egg";
            } else if (currentXP < 300) {
                petDisplay.innerText = "ðŸ›";
                nameDisplay.innerText = "Level 2: The Bug";
            } else if (currentXP < 600) {
                petDisplay.innerText = "ðŸ¤–";
                nameDisplay.innerText = "Level 3: The Bot";
            } else if (currentXP < 1000) {
                petDisplay.innerText = "ðŸ¦„";
                nameDisplay.innerText = "Level 4: The Unicorn";
            } else {
                petDisplay.innerText = "ðŸ‰";
                nameDisplay.innerText = "Level 5: The Dragon";
            }
        }

        // --- CORE LOGIC ---
        function feedPet() {
            const minutes = parseInt(document.getElementById('minutes').value) || 0;
            const lines = parseInt(document.getElementById('lines').value) || 0;

            if (minutes === 0 && lines === 0) {
                alert("Please enter some stats!");
                return;
            }

            // 1. Calculate XP
            const gainedXP = minutes + (lines * 0.5);
            currentXP += gainedXP;

            // 2. Calculate Streak
            checkStreak();

            // 3. Save Data
            saveData();

            // 4. Clear Inputs & Update
            document.getElementById('minutes').value = '';
            document.getElementById('lines').value = '';
            updatePetVisuals();
            
            // Simple Animation trigger
            const pet = document.getElementById('pet-display');
            pet.style.transform = "scale(1.2)";
            setTimeout(() => pet.style.transform = "scale(1)", 200);
        }

        function checkStreak() {
            const today = new Date().toDateString(); // e.g., "Fri Dec 05 2025"
            
            if (lastLogDate === today) {
                // Already logged today, do nothing to streak
                return;
            }

            const yesterday = new Date();
            yesterday.setDate(yesterday.getDate() - 1);
            
            if (lastLogDate === yesterday.toDateString()) {
                // Logged yesterday, so increment streak
                streak++;
            } else {
                // Missed more than one day (or first time)
                streak = 1;
            }
            
            // Update the last log date to today
            lastLogDate = today;
        }

        function saveData() {
            localStorage.setItem('codemonXP', currentXP);
            localStorage.setItem('codemonStreak', streak);
            localStorage.setItem('codemonLastDate', lastLogDate);
        }

        function resetPet() {
            if(confirm("Reset everything? Your pet and streak will be gone.")) {
                localStorage.clear();
                currentXP = 0;
                streak = 0;
                lastLogDate = "";
                updatePetVisuals();
            }
        }
        
        function petClick() {
            alert("Your pet looks happy to see you code!");
        }

        // Initialize on load
        updatePetVisuals();
    </script>
